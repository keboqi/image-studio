# The main WebUI currently depends on the NumPy 1.x binary ABI. OpenCV 4.12+
# requires NumPy 2 on Python 3.12, so keep both OpenCV distributions below it.
numpy<2.0
opencv-python<4.12
opencv-python-headless<4.12
# Pin torch to 2.8.x (matching the nunchaku cu12.8+torch2.8 wheel). Without
# this constraint, subsequent uv_install -U calls pull torch from PyPI and
# upgrade it past the cu128 version, breaking NCCL.
torch<2.9
torchvision<0.24
torchaudio<2.9
